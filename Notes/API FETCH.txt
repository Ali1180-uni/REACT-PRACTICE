Always Fetch Api in the useEffect
  useEffect(() => {
    fetch("https://jsonplaceholder.typicode.com/comments")
      .then((res) => {
        return res.json(); // Must Return the JSON
      })
      .then((resData) => {
        setdata(resData);
      })
      .catch((err) => {
        console.log(err);
      });
  }, []);
